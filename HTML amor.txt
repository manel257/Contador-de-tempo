    <script>
        function atualizarContador() {
            const inicioRelacao = new Date("2023-06-01T00:00:00"); // Data de início do relacionamento
            const agora = new Date();
            const diferenca = agora - inicioRelacao;

            const dias = Math.floor(diferenca / (1000 * 60 * 60 * 24));
            const horas = Math.floor((diferenca % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutos = Math.floor((diferenca % (1000 * 60 * 60)) / (1000 * 60));
            const segundos = Math.floor((diferenca % (1000 * 60)) / 1000);

            document.getElementById("contador").textContent = `${dias} dias ${horas.toString().padStart(2, '0')}:${minutos.toString().padStart(2, '0')}:${segundos.toString().padStart(2, '0')}`;
        }

        setInterval(atualizarContador, 1000);
        atualizarContador();

        // Script para o efeito de fade no carrossel
        let currentIndex = 0;
        const images = document.querySelectorAll('.carousel-container img');
        const totalImages = images.length;

        function showNextImage() {
            images[currentIndex].classList.remove('active'); // Remove a classe 'active' da imagem atual
            currentIndex = (currentIndex + 1) % totalImages; // Passa para a próxima imagem
            images[currentIndex].classList.add('active'); // Adiciona a classe 'active' à nova imagem
        }

        setInterval(showNextImage, 3000); // Troca de imagem a cada 3 segundos
    </script>
